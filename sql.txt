-- 11. TRIGGERS

-- Trigger 1: Create Profile on Auth Signup
create or replace function public.handle_new_user() 
returns trigger as $$
declare
  v_role text;
begin
  v_role := coalesce(new.raw_user_meta_data->>'role', 'client');
  
  insert into public.profiles (id, email, full_name, role, credits)
  values (
    new.id, 
    new.email, 
    coalesce(new.raw_user_meta_data->>'full_name', 'New User'),
    v_role,
    100 -- Default Free Credits
  )
  on conflict (id) do nothing;

  if v_role = 'guruba' then
    insert into public.gurubas (user_id) values (new.id)
    on conflict (user_id) do nothing;
  end if;

  return new;
end;
$$ language plpgsql security definer;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();

-- Trigger 2: Notification & Auto-Message Generator for Bookings
create or replace function public.handle_booking_notification() 
returns trigger as $$
declare
  v_service_title text;
  v_guruba_user_id uuid;
  v_client_name text;
  v_guruba_name text;
begin
  -- 1. Get Details
  select title into v_service_title from public.services where id = NEW.service_id;
  select user_id into v_guruba_user_id from public.gurubas where id = NEW.guruba_id;
  select full_name into v_client_name from public.profiles where id = NEW.user_id;
  select full_name into v_guruba_name from public.profiles where id = v_guruba_user_id;

  -- 2. INSERT: New Booking Request
  if (TG_OP = 'INSERT') then
    -- Notify Guruba
    insert into public.notifications (user_id, title, message)
    values (
      v_guruba_user_id, 
      'New Booking Request', 
      'New request for ' || v_service_title || ' from ' || coalesce(v_client_name, 'a client')
    );
    -- Chat Message: Client -> Guruba
    insert into public.messages (sender_id, receiver_id, content, created_at)
    values (
       NEW.user_id,
       v_guruba_user_id,
       'Namaste, I have requested a booking for ' || v_service_title || '. Please review.',
       now()
    );
  end if;

  -- 3. UPDATE: Status Changes
  if (TG_OP = 'UPDATE' and OLD.status is distinct from NEW.status) then
    
    -- Confirmed -> Notify Client
    if (NEW.status = 'confirmed') then
      insert into public.notifications (user_id, title, message)
      values (
        NEW.user_id, 
        'Booking Confirmed', 
        'Your request for ' || v_service_title || ' has been confirmed by ' || coalesce(v_guruba_name, 'the Guruba')
      );
       -- Chat Message: Guruba -> Client
       insert into public.messages (sender_id, receiver_id, content, created_at)
       values (
          v_guruba_user_id,
          NEW.user_id,
          'I have confirmed your booking for ' || v_service_title || '. Looking forward to it.',
          now()
       );
    end if;

    -- Proposed Time -> Notify Client
    if (NEW.status = 'awaiting_client_confirmation') then
      insert into public.notifications (user_id, title, message)
      values (
        NEW.user_id, 
        'Time Change Proposed', 
        'Guruba has proposed a new time for ' || v_service_title || '. Please accept within 1 hour.'
      );
       -- Chat Message: Guruba -> Client
       insert into public.messages (sender_id, receiver_id, content, created_at)
       values (
          v_guruba_user_id,
          NEW.user_id,
          'I am available, but I propose a slight time change to: ' || NEW.proposed_time || '. Please confirm if this works.',
          now()
       );
    end if;

    -- Cancelled -> Notify Both (Generic)
    if (NEW.status = 'cancelled') then
      insert into public.notifications (user_id, title, message)
      values (NEW.user_id, 'Booking Cancelled', 'The booking for ' || v_service_title || ' has been cancelled.');
      insert into public.notifications (user_id, title, message)
      values (v_guruba_user_id, 'Booking Cancelled', 'The booking for ' || v_service_title || ' has been cancelled.');
    end if;

    -- Completed -> Notify Client
    if (NEW.status = 'completed') then
      insert into public.notifications (user_id, title, message)
      values (NEW.user_id, 'Service Completed', 'Your ' || v_service_title || ' is complete. Please leave a review!');
    end if;

  end if;

  return NEW;
end;
$$ language plpgsql security definer;

drop trigger if exists on_booking_change on public.bookings;
create trigger on_booking_change
  after insert or update on public.bookings
  for each row execute procedure public.handle_booking_notification();

-- Trigger 3: Sync guruba_services to legacy specialties array
create or replace function public.sync_specialties()
returns trigger as $$
begin
  -- Update the legacy text array in gurubas table based on currently selected services
  update public.gurubas
  set specialties = array(
    select s.title
    from public.guruba_services gs
    join public.services s on s.id = gs.service_id
    where gs.guruba_id = coalesce(NEW.guruba_id, OLD.guruba_id)
  )
  where id = coalesce(NEW.guruba_id, OLD.guruba_id);
  return NEW;
end;
$$ language plpgsql security definer;

drop trigger if exists on_guruba_service_change on public.guruba_services;
create trigger on_guruba_service_change
  after insert or update or delete on public.guruba_services
  for each row execute procedure public.sync_specialties();

-- 14. RPC FUNCTIONS (New)
-- Function to book service and deduct credits safely
create or replace function public.book_service(
  p_user_id uuid,
  p_guruba_id uuid,
  p_service_id uuid,
  p_scheduled_at timestamp with time zone,
  p_platform_fee integer
)
returns uuid
language plpgsql
security definer
as $$
declare
  v_booking_id uuid;
  v_current_credits integer;
begin
  -- 1. Check credits
  select credits into v_current_credits from public.profiles where id = p_user_id;
  
  if v_current_credits < p_platform_fee then
    raise exception 'Insufficient credits';
  end if;

  -- 2. Deduct credits
  update public.profiles 
  set credits = credits - p_platform_fee 
  where id = p_user_id;

  -- 3. Create booking
  insert into public.bookings (user_id, guruba_id, service_id, scheduled_at, status, platform_fee)
  values (p_user_id, p_guruba_id, p_service_id, p_scheduled_at, 'pending', p_platform_fee)
  returning id into v_booking_id;

  -- 4. Record transaction
  insert into public.transactions (user_id, amount, type, description, status)
  values (p_user_id, p_platform_fee, 'debit', 'Booking Fee for Service', 'completed');

  return v_booking_id;
end;
$$;
